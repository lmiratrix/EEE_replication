for INDEX_VAR1 in $(seq 1 10); do
 for INDEX_VAR2 in $(seq 1 10); do

 #print out indexes
 echo “${INDEX_VAR1}, ${INDEX_VAR2}”

#give these indexes to R
 export INDEX_VAR1 INDEX_VAR2

 #Run R script, and produce output files
 sbatch -o out_p${INDEX_VAR1}_s${INDEX_VAR2}.stdout.txt \
  -e out_p${INDEX_VAR1}_s${INDEX_VAR2}.stdout.txt \
  --job-name=runScript_p${INDEX_VAR1} \
  sbatch_runScript.txt
  #
  sleep 1 # pause to be kind to the scheduler
 done
done
